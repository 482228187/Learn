var addStrings = function(num1, num2) {
    let i=num1.length-1 , j=num2.length-1
    let carry = 0
    let ans = []
    while(j>=0 || i>=0 || carry!==0){
        let c1 = i>=0 ? +num1.charAt(i--) : 0,
        c2 = j >= 0 ? +num2.charAt(j--)   : 0;
